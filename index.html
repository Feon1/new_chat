const ADAPTER_ROOT = 'https://new-chat-adapter.onrender.com'; // без /mcp const ADAPTER_URL = ADAPTER_ROOT + '/mcp'; function pingAdapter(cb) { // Проверяем доступность корневого пути (он возвращает 200 OK) const xhr = new XMLHttpRequest(); xhr.open('GET', ADAPTER_ROOT, true); xhr.timeout = 5000; xhr.onload = function() { if (this.status === 200) cb(true); else cb(false); }; xhr.onerror = function() { cb(false); }; xhr.ontimeout = function() { cb(false); }; try { xhr.send(); } catch(e) { cb(false); } }
